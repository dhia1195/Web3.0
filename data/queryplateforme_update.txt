  ?platform myOnto:nomPlateform ?oldName .

  ?platform myOnto:nomPlateform ?newName .
  ?platform myOnto:typeAbonnement ?newSubscriptionType .

  BIND(<http://www.semanticweb.org/emnar/ontologies/2024/9/untitled-ontology-7#Plateforme_SOME_UNIQUE_ID> AS ?platform)
  ?platform myOnto:nomPlateform ?oldName .
  ?platform myOnto:typeAbonnement ?oldSubscriptionType .

  BIND("New Subscription Type" AS ?newSubscriptionType)  # Replace with the actual new subscription type
